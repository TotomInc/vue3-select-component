The `options` prop populates the dropdown menu.

## Use the basic structure

Each option needs a `label` and a `value`:

```vue [App.vue]
<script setup lang="ts">
import { ref } from "vue"
import Select from "vue3-select-component"
import "vue3-select-component/styles"

const selected = ref<string | null>(null)
</script>

<template>
  <Select
    v-model="selected"
    :options="[
      { label: 'JavaScript', value: 'js' },
      { label: 'TypeScript', value: 'ts' },
    ]"
    placeholder="Pick a language"
  />
</template>
```

| Property | Description |
|----------|-------------|
| `label` | Text shown in the dropdown and selected value |
| `value` | Data bound to `v-model` when selected (`string` or `number`) |

::note
Import the `SelectOption` type (exported as `SelectOptionData`) for TypeScript:

```ts
import type { SelectOptionData } from "vue3-select-component"

const options: SelectOptionData<string>[] = [
  { label: "JavaScript", value: "js" },
]
```
::

## Disable individual options

Add `disabled: true` to prevent selection:

```vue [App.vue]
<script setup lang="ts">
import { ref } from "vue"
import Select from "vue3-select-component"
import "vue3-select-component/styles"

const selected = ref<string | null>(null)
</script>

<template>
  <Select
    v-model="selected"
    :options="[
      { label: 'Available', value: 'a1' },
      { label: 'Unavailable', value: 'a2', disabled: true },
    ]"
    placeholder="Pick an option"
  />
</template>
```

Disabled options cannot be selected. They are skipped during keyboard navigation and get `aria-disabled`.

## Use custom option shapes

When your data does not match `{ label, value }`, provide mappers:

```vue [App.vue]
<script setup lang="ts">
import { ref } from "vue"
import Select from "vue3-select-component"
import "vue3-select-component/styles"

type Country = { id: string; key: string }

const countries: Country[] = [
  { id: "France", key: "fr" },
  { id: "Germany", key: "de" },
]

const selected = ref<string | null>(null)
</script>

<template>
  <Select
    v-model="selected"
    :options="countries"
    :get-option-label="(o) => o.id"
    :get-option-value="(o) => o.key"
    placeholder="Select a country"
  />
</template>
```

| Prop | Type | Default |
|------|------|---------|
| `getOptionLabel` | `(option) => string` | `option.label` |
| `getOptionValue` | `(option) => string \| number` | `option.value` |

## Customize search filtering

By default, search matches case-insensitive substrings on the option label.

**Assembled `Select`**: filter receives the source option object:

```ts
type GenericFilterByFn<GenericOption> = (
  option: GenericOption,
  label: string,
  search: string,
) => boolean
```

```vue [App.vue]
<script setup lang="ts">
import { ref } from "vue"
import Select from "vue3-select-component"
import "vue3-select-component/styles"

type CountryOption = {
  label: string
  value: string
  code: string
}

const selected = ref<string | null>(null)

const options: CountryOption[] = [
  { label: "France", value: "FR", code: "fr" },
  { label: "Germany", value: "DE", code: "de" },
]
</script>

<template>
  <Select
    v-model="selected"
    :options="options"
    :filter-by="(option, label, search) =>
      label.toLowerCase().includes(search.toLowerCase())
      || option.code.toLowerCase().includes(search.toLowerCase())"
    placeholder="Search by name or code"
  />
</template>
```

**Primitives (`SelectRoot`)**: filter receives the registered option:

```ts
type FilterByFn<Value> = (
  option: RegisteredOption<Value>,
  search: string,
) => boolean
```

## Add extended properties

Options can carry extra fields for custom rendering via primitive slots:

```vue [CustomSelect.vue]
<script setup lang="ts">
import { ref } from "vue"
import {
  SelectInput,
  SelectListbox,
  SelectNoOptions,
  SelectOption,
  SelectPopover,
  SelectRoot,
  SelectTrailingIcon,
  SelectTrigger,
  SelectValue,
} from "vue3-select-component/primitives"
import "vue3-select-component/styles"

type LanguageOption = {
  label: string
  value: string
  version: string
  creator: string
}

const selected = ref<string | null>(null)

const options: LanguageOption[] = [
  { label: "JavaScript", value: "js", version: "ES2022", creator: "Brendan Eich" },
  { label: "TypeScript", value: "ts", version: "5.0", creator: "Microsoft" },
]
</script>

<template>
  <div data-assembled-select>
    <SelectRoot v-model="selected" :options="options" searchable>
      <SelectTrigger>
        <SelectValue placeholder="Pick a language" />
        <SelectInput />
        <SelectTrailingIcon />
      </SelectTrigger>

      <SelectPopover>
        <SelectListbox>
          <SelectNoOptions />
          <SelectOption
            v-for="option in options"
            :key="option.value"
            :value="option.value"
            :label="option.label"
          >
            <span>{{ option.label }} ({{ option.version }})</span>
            <span>{{ option.creator }}</span>
          </SelectOption>
        </SelectListbox>
      </SelectPopover>
    </SelectRoot>
  </div>
</template>
```

See [TypeScript](/guide/typescript) for extending option types safely.
