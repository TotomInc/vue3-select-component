Vue 3 Select Component is built with TypeScript for full type safety and IDE support.

## Use the SelectOption type

The canonical option shape is exported as `SelectOptionData`:

```ts
import type { SelectOptionData } from "vue3-select-component"

type SelectOption<Value extends string | number = string> = {
  label: string
  value: Value
  disabled?: boolean
}
```

## Configure value types

`option.value` and `v-model` share the same type. Use `string` or `number`:

```vue [App.vue]
<script setup lang="ts">
import type { SelectOptionData } from "vue3-select-component"
import { ref } from "vue"
import Select from "vue3-select-component"

type UserOption = SelectOptionData<number>

const selectedUser = ref<number | null>(null)

const userOptions: UserOption[] = [
  { label: "Alice", value: 1 },
  { label: "Bob", value: 2 },
  // { label: "Charlie", value: "invalid" } // Type error
]
</script>

<template>
  <Select
    v-model="selectedUser"
    :options="userOptions"
    placeholder="Pick a user"
  />
</template>
```

## Extend option properties

Add custom fields with mappers and primitive slots:

```vue [App.vue]
<script setup lang="ts">
import type { SelectOptionData } from "vue3-select-component"
import { ref } from "vue"
import Select from "vue3-select-component"

type UserOption = SelectOptionData<number> & { username: string }

const selectedUser = ref<number | null>(null)

const userOptions: UserOption[] = [
  { label: "Alice", value: 1, username: "alice15" },
  { label: "Bob", value: 2, username: "bob01" },
]
</script>

<template>
  <Select
    v-model="selectedUser"
    :options="userOptions"
    :get-option-label="(option) => `${option.label} (${option.username})`"
    placeholder="Pick a user"
  />
</template>
```

For custom option rendering, compose with primitives and use the `SelectOption` slot.

## Validate v-model types

The component enforces consistency between `option.value` and `v-model`:

```vue
<script setup lang="ts">
import type { SelectOptionData } from "vue3-select-component"
import { ref } from "vue"

type UserOption = SelectOptionData<number>

// Incorrect: should be ref<number | null>(null)
const selectedUser = ref<string | null>(null)

const userOptions: UserOption[] = [
  { label: "Alice", value: 1 },
]
</script>
```

## Use SelectModelValue

The `v-model` type is `SelectModelValue<Value>`:

```ts
import type { SelectModelValue } from "vue3-select-component"

// Single: Value | null
// Multi:  Value[]
type SingleModel = SelectModelValue<string>   // string | string[] | null
```

Bind `ref<string | null>(null)` for single select and `ref<string[]>([])` for multi-select. TypeScript infers the narrow type from usage.

## Import component prop types

```ts
import type {
  AssembledSelectProps,
  AssembledSelectEmits,
  SelectPopoverProps,
  SelectRootProps,
  SelectRootEmits,
} from "vue3-select-component"
```

Use these when wrapping the component or building higher-level abstractions.
