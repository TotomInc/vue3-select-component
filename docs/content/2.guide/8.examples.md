## Assembled select

The default export wires up primitives with sensible defaults. Use these patterns when the built-in UX is enough.

### Clearable

Show a clear button when a value is selected.

::client-only
:example-clearable-demo

#fallback
:select-demo-fallback
::

```vue [App.vue]
<script setup lang="ts">
import { ref } from "vue"
import Select from "vue3-select-component"
import "vue3-select-component/styles"

const selected = ref<string | null>("FR")

const options = [
  { label: "France", value: "FR" },
  { label: "Spain", value: "ES" },
  { label: "United Kingdom", value: "GB" },
  { label: "Germany", value: "DE" },
  { label: "Italy", value: "IT" },
]
</script>

<template>
  <Select
    v-model="selected"

    placeholder="Select a country"
```
```vue [App.vue]
  />
</template>
```

## Primitives

Compose `SelectRoot`, `SelectTrigger`, `SelectPopover`, and the rest when you need custom layout, slots, or popover behavior. Demos use `data-assembled-select` on a wrapper so default styles apply while you own the DOM tree.

### Full composition

Wire every trigger and listbox primitive yourself. Add a leading icon, search input, clear button, and a custom trailing icon.

::client-only
:example-primitive-composed-demo

#fallback
:select-demo-fallback
::

```vue [CustomSelect.vue]
<script setup lang="ts">
import { ref } from "vue"
import {
  SelectClear,
  SelectIcon,
  SelectInput,
  SelectListbox,
  SelectNoOptions,
  SelectOption,
  SelectPopover,
  SelectRoot,
  SelectTrailingIcon,
  SelectTrigger,
  SelectValue,
} from "vue3-select-component/primitives"
import "vue3-select-component/styles"

const selected = ref<string | null>(null)

const options = [
  { label: "France", value: "FR" },
  { label: "Spain", value: "ES" },
  { label: "Germany", value: "DE" },
]
</script>

<template>
  <div data-assembled-select>
    <SelectRoot
      v-model="selected"
      searchable
      clearable
    >
      <SelectTrigger>
        <SelectIcon aria-hidden="true">
          <!-- globe icon -->
        </SelectIcon>
        <SelectValue placeholder="Select a country" />
        <SelectInput />
        <SelectTrailingIcon>
          <template #default="{ open }">
            <span aria-hidden="true">{{ open ? "▲" : "▼" }}</span>
          </template>
        </SelectTrailingIcon>
        <SelectClear />
      </SelectTrigger>

      <SelectPopover>
        <SelectListbox>
          <SelectNoOptions />
          <SelectOption
            v-for="option in options"
            :key="option.value"
            :value="option.value"
            :label="option.label"
          />
        </SelectListbox>
      </SelectPopover>
    </SelectRoot>
  </div>
</template>
```

### Create item

Let users add a value from the search query. Handle `@create` to push into your options list and update `v-model`.

::client-only

:select-demo-fallback
::

```vue [CustomSelect.vue]
<script setup lang="ts">
import { ref } from "vue"
import {
  SelectCreateItem,
  SelectInput,
  SelectListbox,
  SelectNoOptions,
  SelectOption,
  SelectPopover,
  SelectRoot,
  SelectTrailingIcon,
  SelectTrigger,
  SelectValue,
} from "vue3-select-component/primitives"
import "vue3-select-component/styles"

const statuses = ref(["Backlog", "Todo", "In Progress", "Done"])
const selected = ref("Backlog")

function onCreate(item: string) {
  statuses.value.push(item)
  selected.value = item
}
</script>

<template>
  <div data-assembled-select>
    <SelectRoot
      v-model="selected"
      searchable
      create-item
      @create="onCreate"
    >
      <SelectTrigger>
        <SelectValue placeholder="Pick or create a status" />
        <SelectInput />
        <SelectTrailingIcon />
      </SelectTrigger>

      <SelectPopover>
        <SelectListbox>
          <SelectNoOptions />
          <SelectOption
            v-for="status in statuses"
            :key="status"
            :value="status"
            :label="status"
          />
          <SelectCreateItem>
            <template #default="{ searchValue }">
              Add "{{ searchValue }}"
            </template>
          </SelectCreateItem>
        </SelectListbox>
      </SelectPopover>
    </SelectRoot>
  </div>
</template>
```

### Filter by extra fields

Pass a custom `filterBy` to `SelectRoot` and render extra metadata in each `SelectOption` slot.

::client-only
:example-primitive-region-filter-demo

#fallback
:select-demo-fallback
::

```vue [CustomSelect.vue]
<script setup lang="ts">
import { ref } from "vue"
import {
  SelectInput,
  SelectListbox,
  SelectNoOptions,
  SelectOption,
  SelectPopover,
  SelectRoot,
  SelectTrailingIcon,
  SelectTrigger,
  SelectValue,
} from "vue3-select-component/primitives"
import "vue3-select-component/styles"

type CountryRecord = {
  name: string
  code: string
  region: string
}

const selected = ref<string | null>(null)

const countries: CountryRecord[] = [
  { name: "France", code: "FR", region: "Europe" },
  { name: "Japan", code: "JP", region: "Asia" },
  { name: "Brazil", code: "BR", region: "South America" },
  { name: "Canada", code: "CA", region: "North America" },
]

function filterByRegion(
  option: { label: string; value: string },
  search: string,
) {
  const query = search.trim().toLowerCase()

  if (!query) {
    return true
  }

  const record = countries.find((entry) => entry.code === option.value)

  return option.label.toLowerCase().includes(query)
    || record?.region.toLowerCase().includes(query) === true
}
</script>

<template>
  <div data-assembled-select>
    <SelectRoot
      v-model="selected"
      searchable
      :filter-by="filterByRegion"
    >
      <SelectTrigger>
        <SelectValue placeholder="Search by country or region" />
        <SelectInput />
        <SelectTrailingIcon />
      </SelectTrigger>

      <SelectPopover>
        <SelectListbox>
          <SelectNoOptions />
          <SelectOption
            v-for="country in countries"
            :key="country.code"
            :value="country.code"
            :label="country.name"
          >
            <span>{{ country.name }}</span>
            <span>{{ country.region }}</span>
          </SelectOption>
        </SelectListbox>
      </SelectPopover>
    </SelectRoot>
  </div>
</template>
```
